    <!-- JAVASCRIPT LOGIC -->
    <script>
        // --- 1. Music & Intro Logic ---
        const overlay = document.getElementById('intro-overlay');
        const mainContent = document.getElementById('main-content');
        const audio = document.getElementById('bg-music');
        const musicBtn = document.getElementById('music-toggle');
        let isPlaying = false;

        // User must interact to play audio (Browser Policy)
        overlay.addEventListener('click', () => {
            // Fade out overlay
            gsap.to(overlay, { opacity: 0, duration: 1, onComplete: () => overlay.style.display = 'none' });
            
            // Show main content
            mainContent.classList.remove('hidden');
            gsap.to(mainContent, { opacity: 1, duration: 1.5 });

            // Play Audio
            audio.volume = 0.5;
            audio.play().then(() => {
                isPlaying = true;
                musicBtn.innerHTML = '<i class="fa-solid fa-pause"></i>';
                startAnimations(); // Start GSAP animations
            }).catch(err => console.log("Audio play blocked", err));
        });

        // Music Toggle Logic
        musicBtn.addEventListener('click', () => {
            if (isPlaying) {
                audio.pause();
                musicBtn.innerHTML = '<i class="fa-solid fa-music"></i>';
                musicBtn.classList.remove('animate-spin-slow');
            } else {
                audio.play();
                musicBtn.innerHTML = '<i class="fa-solid fa-pause"></i>';
            }
            isPlaying = !isPlaying;
        });

        // --- 2. GSAP Animations ---
        function startAnimations() {
            const tl = gsap.timeline();

            // Hero Animations
            tl.from("#profile-img", { scale: 0, rotation: 360, duration: 1.5, ease: "elastic.out(1, 0.5)" })
              .from(".greeting-text", { y: 20, opacity: 0, duration: 0.8 }, "-=1")
              .from(".title-text", { y: 30, opacity: 0, duration: 1, ease: "power3.out" }, "-=0.5")
              .from(".message-text", { y: 20, opacity: 0, duration: 0.8 }, "-=0.5");

            // Scroll Trigger for Wishes (Simple Intersection Observer wrapper)
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        gsap.to(".wish-card", {
                            y: 0,
                            opacity: 1,
                            duration: 0.8,
                            stagger: 0.2,
                            ease: "power2.out"
                        });
                    }
                });
            }, { threshold: 0.1 });

            const wishesSection = document.getElementById('wishes-section');
            if(wishesSection) observer.observe(wishesSection);

            // Start Heart Generator
            setInterval(createHeart, 800);
        }

        function scrollToWishes() {
            document.getElementById('wishes-section').scrollIntoView({ behavior: 'smooth' });
        }

        // --- 3. Floating Glowing Hearts System ---
        function createHeart() {
            const container = document.getElementById('hearts-container');
            const heart = document.createElement('div');
            
            // Random icons (Heart or Star)
            const iconType = Math.random() > 0.7 ? 'fa-star' : 'fa-heart';
            heart.innerHTML = `<i class="fa-solid ${iconType}"></i>`;
            
            heart.classList.add('heart');
            
            // Random positioning
            const startLeft = Math.random() * 100;
            const size = Math.random() * 20 + 10; // 10px to 30px
            const duration = Math.random() * 3 + 4; // 4s to 7s float time
            
            heart.style.left = startLeft + 'vw';
            heart.style.fontSize = size + 'px';
            heart.style.bottom = '-50px';
            heart.style.opacity = Math.random() * 0.5 + 0.2;
            
            container.appendChild(heart);

            // Animate with GSAP
            gsap.to(heart, {
                y: -window.innerHeight - 100,
                x: (Math.random() - 0.5) * 200, // Random wobble left/right
                rotation: Math.random() * 360,
                duration: duration,
                ease: "none",
                onComplete: () => {
                    heart.remove();
                }
            });
       
 }
    </script>
